C++中未使用虚析构函数导致的内存泄漏

代码

#include <vector>
#include <iostream>
#include "vld.h"

using namespace std;

class Base
{
public:
    Base(){}
    ~Base()
    {
        cout << "Base Destructor" << endl;
    }
};

class Derive : public Base
{
public:
    Derive()
    {
        vecInt.push_back(0);
        vecInt.push_back(2);
    }
    ~Derive()
    {
        cout << "Derive Destructor" << endl;
    }

    vector<int> vecInt;
};

int main(int argc, char ** argv)
{
    Base * a = new Derive();
    delete a;
    return 0;
}
