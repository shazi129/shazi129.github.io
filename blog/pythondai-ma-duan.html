下面总结一些日常Python代码段

遍历文件夹

import os
def getAllFile(dirPath):
    ret = []
    allsub = os.listdir(dirPath)
    for sub in allsub:
        subPath = "%s/%s" % (dirPath, sub)
        if os.path.isfile(subPath):
            ret.append(subPath)
        if os.path.isdir(subPath):
            ret.extend(getAllFile(subPath))
    return ret
